1   package nl.tochbedrijf.frontoffice.services;
2   
3   import java.util.List;
4   import java.util.Optional;
5   import java.util.UUID;
6   import java.util.stream.Collectors;
7   import nl.tochbedrijf.frontoffice.entities.Book;
8   import nl.tochbedrijf.frontoffice.repositories.BookRepository;
9   import nl.tochbedrijf.frontoffice.services.dtos.BookDTO;
10  import org.springframework.stereotype.Service;
11  
12  @Service
13  public class BookService {
14  
15    private final BookRepository bookRepository;
16  
17    public BookService(BookRepository bookRepository) {
18      this.bookRepository = bookRepository;
19    }
20  
21    public List<BookDTO> getBooks() {
22      return bookRepository.findAll().stream().map(this::convertToDto).collect(Collectors.toList());
23    }
24  
25    public BookDTO getBookById(Long id) {
26      Optional<Book> book = bookRepository.findById(id);
27      if (book.isPresent()) {
28        return convertToDto(book.get());
29      } else {
30        throw new RuntimeException("Book not found");
31      }
32    }
33  
34    public BookDTO save(BookDTO bookDTO) {
35      return convertToDto(bookRepository.save(convertToEntity(bookDTO)));
36    }
37  
38    public void deleteId(Long id) {
39      bookRepository.deleteById(id);
40    }
41  
42    public BookDTO convertToDto(Book book) {
43      return new BookDTO(book.getId(), book.getTitle(), book.getAuthor());
44    }
45  
46    public Book convertToEntity(BookDTO bookDTO) {
47      return new Book(
48          bookDTO.getId(), bookDTO.getTitle(), bookDTO.getAuthor(), UUID.randomUUID().toString());
49    }
50  
51    public BookDTO updateBook(Long id, BookDTO bookDTO) {
52      Optional<Book> toBoUpdated = bookRepository.findById(id);
53      if (toBoUpdated.isPresent()) {
54        Book book = toBoUpdated.get();
55        book.setTitle(bookDTO.getTitle());
56        book.setAuthor(bookDTO.getAuthor());
57        return convertToDto(bookRepository.save(book));
58  
59      } else {
60        throw new RuntimeException("Update failed, book not found");
61      }
62    }
63  }
